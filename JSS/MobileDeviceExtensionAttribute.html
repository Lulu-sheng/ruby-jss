def input_type= (new_val)
  raise JSS::InvalidDataError, "Mobile Device Extension Attribute input_type cannot be 'script'" if new_val == 'script'

  super

  if @input_type == 'LDAP Attribute Mapping'
    @popup_choices = nil
  else
    @attribute_mapping = nil
  end
end
